

# library
library(dplyr)
library(reshape2)
library(ggplot2)

# load
load("3_ARG_taxonomy_mapping_total_df.Rdata")

remove(ARG_Class_mapping_total_df, ARG_Family_mapping_total_df, ARG_Genus_mapping_total_df, ARG_Order_mapping_total_df)

ARG_Phylum_mapping_total_df$sampleType <- sapply(strsplit(ARG_Phylum_mapping_total_df$sample_name, "_", fixed = T), "[[", 2)


# 1.цек╝ф╫
# 
ARG_Phylum_mapping_total_df <- ARG_Phylum_mapping_total_df %>% filter(Phylum != "Others")
ARG_Phylum_mapping_total_df <- ARG_Phylum_mapping_total_df %>% filter(sampleName != "ZJ.T5")
ARG_Phylum_mapping_total_df <- ARG_Phylum_mapping_total_df %>% filter(sampleName != "ZJ.T9")

Total_phylum_df <- ARG_Phylum_mapping_total_df %>% group_by(sampleName) %>% summarise(DepthPG = sum(DepthPG)) %>% as.data.frame()
#

ARG_Phylum_mapping_total_df$total_Phylum <- sapply(ARG_Phylum_mapping_total_df$sampleName, 
                                       function(x) Total_phylum_df$DepthPG[which(Total_phylum_df$sampleName == x)])

ARG_Phylum_mapping_total_df$Phylum_percent <- ARG_Phylum_mapping_total_df$DepthPG/ARG_Phylum_mapping_total_df$total_Phylum

sum(ARG_Phylum_mapping_total_df$Phylum_percent) #352

# 

Phylum_relAbund_df <- ARG_Phylum_mapping_total_df %>% dplyr::group_by(Phylum) %>%
  dplyr::summarise(meanpercent=mean(Phylum_percent)) %>%
  as.data.frame() %>% 
  arrange(desc(meanpercent))

otherTaxonomy <- Phylum_relAbund_df$Phylum[11:nrow(Phylum_relAbund_df)] 

#
ARG_Phylum_mapping_total_df$Phylum[ARG_Phylum_mapping_total_df$Phylum %in% otherTaxonomy] <- "Others"

ARG_Phylum_mapping_total_df <- ARG_Phylum_mapping_total_df %>% group_by(sampleName, Phylum) %>% 
  summarise(Phylum_percent=sum(Phylum_percent), sampleType = unique(sampleType))

#
ARG_Phylum_mapping_total_df_width <- dcast(ARG_Phylum_mapping_total_df, sampleName+sampleType~ARG_Phylum_mapping_total_df$Phylum, value.var = "Phylum_percent")
ARG_Phylum_mapping_total_df_width[is.na(ARG_Phylum_mapping_total_df_width)] <- 0

ARG_Phylum_mapping_total_df <- melt(ARG_Phylum_mapping_total_df_width, variable.name = "Phylum", value.name = "Phylum_percent") #3872

str(ARG_Phylum_mapping_total_df) # factor
ARG_Phylum_mapping_total_df$Phylum <- as.character(ARG_Phylum_mapping_total_df$Phylum)

# 
ARG_Phylum_mapping_total_df <- ARG_Phylum_mapping_total_df %>% group_by(sampleType, Phylum) %>% 
  summarise(Phylum_percent=mean(Phylum_percent))

sum(ARG_Phylum_mapping_total_df$Phylum_percent)

# plot

ARG_Phylum_mapping_total_df$Phylum <- factor(ARG_Phylum_mapping_total_df$Phylum, 
                                  levels = c("Others", "Candidatus Rokubacteria", "Verrucomicrobia", "Bacteroidota","Fusobacteria",
                                             "Gemmatimonadetes","Firmicutes","Chloroflexi","Acidobacteria","Actinobacteria","Proteobacteria")) 

ARG_Phylum_mapping_total_df$sampleType <- factor(ARG_Phylum_mapping_total_df$sampleType, levels = c("tailings","Sludge","farmland","forest","grass","gobi"))

p_ARG_phylum <- ggplot(ARG_Phylum_mapping_total_df, aes(x = sampleType, y = Phylum_percent, fill=Phylum))  + geom_col(width = 0.6) +  
  #scale_fill_manual(values = c("#9FCD99", "#FFDD71","#DFC27D", "#ce3d36", "#e9a91f", "#D9F0D3", "#D1E5F0", "#5c6bc0","#FFB977","#B2AD8F","gray")) + 
  scale_fill_manual(values = c("gray","#6eafd4","#bbdefb","#b67b40", "#fb8d46", "#fcbd89", "#3a9693", "#c5e5e0","#7fb7ae","#f8eac8","#ddc17f")) + 
  theme(axis.text.x = element_text(hjust = 0.5, vjust = 0.5)) + 
  theme_bw(base_size = 20) +
  scale_y_continuous(expand = c(0.0001, 0.015)) +
  scale_x_discrete(expand = c(0.12, 0.02)) +
  xlab("") + ylab("Phyla composition of ARGs") + #+ guides(fill = guide_legend(nrow=2)) 
  theme(legend.position="right") +
  theme(panel.grid.major = element_blank(),
        panel.grid.minor = element_blank(),
        panel.background = element_blank()) + 
  scale_x_discrete(labels = c("tailings" = "Tailings", "Sludge" = "Sewage", "farmland" = "Farmland", 
                              "forest" = "Forest","grass" = "Grass","gobi" = "Gobi"))# + coord_flip() +

p_ARG_phylum +labs(fill="")
